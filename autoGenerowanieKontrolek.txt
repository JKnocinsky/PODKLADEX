        private int ObliczLiczbeSztuk(int idNarzedzia)
        {
            Narzedzie narzedzie = context.Narzedzie.Where(n => n.IdNarzedzia == idNarzedzia).FirstOrDefault();
            int liczbaSztuk = narzedzie.LiczbaPoczatkowa;
            List<UtylizacjaNarzedzia> utylizacje = context.UtylizacjaNarzedzia.Where(u => u.IdNarzedzia == idNarzedzia).ToList();
            List<ZakupNarzedzia> zakupy = context.ZakupNarzedzia.Where(z => z.IdNarzedzia == idNarzedzia).ToList();
            foreach (var utylizacja in utylizacje)
            {
                liczbaSztuk -= utylizacja.LiczbaSztuk;
            }
            foreach (var zakup in zakupy)
            {
                liczbaSztuk += zakup.LiczbaSztuk;
            }
            return liczbaSztuk;
        }
